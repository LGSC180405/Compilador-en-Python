// Programa de prueba léxica
int numero = 10;
float promedio = 3.14;
string mensaje = "Hola mundo!";

/* Bucle de ejemplo */
while (numero > 0) {
    numero = numero - 1;
    promedio = promedio + 0.5;
    mensaje = "Iteración: " + numero;
}

if (promedio >= 5.0) {
    print(mensaje);
} else {
    print("Promedio bajo");
}
